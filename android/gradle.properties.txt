# android/gradle.properties
# Configuration optimisée pour systèmes avec 4GB RAM ou moins

# ESSENTIEL: Limiter la mémoire JVM
org.gradle.jvmargs=-Xmx1g -XX:MaxMetaspaceSize=256m -XX:+UseSerialGC -XX:+HeapDumpOnOutOfMemoryError

# Désactiver le daemon pour éviter les crashes
org.gradle.daemon=false

# Désactiver les processus parallèles (consomment trop de RAM)
org.gradle.parallel=false

# Désactiver le cache (utilise moins de mémoire)
org.gradle.caching=false
org.gradle.configureondemand=false

# Configuration Android standard
android.useAndroidX=true
android.enableJetifier=true

# Désactiver R8 full mode (moins de RAM)
android.enableR8.fullMode=false

# Réduire les ressources Kotlin
kotlin.incremental=false
kotlin.incremental.android=false
